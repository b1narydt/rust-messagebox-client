#[test]
fn parity_05_wire_format_json() {
    // Test serialization of every request/response type
    // Verify camelCase field names match TS wire format

    // SendMessageParams
    let params = SendMessageParams { recipient: "abc".into(), message_box: "inbox".into(), body: "hello".into(), message_id: "id1".into() };
    let json = serde_json::to_value(&params).unwrap();
    assert!(json.get("messageBox").is_some(), "must use camelCase messageBox");
    assert!(json.get("messageId").is_some(), "must use camelCase messageId");

    // AcknowledgeMessageParams
    let ack = AcknowledgeMessageParams { message_ids: vec!["a".into()] };
    let json = serde_json::to_value(&ack).unwrap();
    assert!(json.get("messageIds").is_some());

    // SendListParams
    let slp = SendListParams { recipients: vec!["a".into()], message_box: "inbox".into(), body: "hi".into(), skip_encryption: None };
    let json = serde_json::to_value(&slp).unwrap();
    assert!(json.get("messageBox").is_some());
    assert!(json.get("skipEncryption").is_none(), "None should be skipped");
