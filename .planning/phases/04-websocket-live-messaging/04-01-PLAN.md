```rust
use std::collections::{HashMap, HashSet};
use std::sync::atomic::{AtomicBool, Ordering};
use std::sync::Arc;

use futures_util::FutureExt;
use rust_socketio::asynchronous::{Client as SocketClient, ClientBuilder};
use rust_socketio::{Event, Payload};
use serde_json::json;
use tokio::sync::{oneshot, Mutex};

use bsv::remittance::types::PeerMessage;
use bsv::wallet::interfaces::WalletInterface;

use crate::error::MessageBoxError;
use crate::types::ServerPeerMessage;

pub struct MessageBoxWebSocket {
    client: SocketClient,
    subscriptions: Arc<Mutex<HashMap<String, Arc<dyn Fn(PeerMessage) + Send + Sync>>>>,
    pending_acks: Arc<Mutex<HashMap<String, oneshot::Sender<bool>>>>,
    joined_rooms: Arc<Mutex<HashSet<String>>>,
    connected: Arc<AtomicBool>,
}
```

Implement:

- **`pub async fn connect<W>(url: &str, identity_key: &str, wallet: W, originator: Option<String>) -> Result<Self, MessageBoxError>`** where `W: WalletInterface + Clone + Send + Sync + 'static`:

  1. Create shared state: `subscriptions`, `pending_acks`, `connected` as `Arc`s.
  2. Create `oneshot::channel::<bool>()` for auth signal.
  3. Build `ClientBuilder::new(url)` with `.on_any(...)` callback:
     - Clone all Arcs + wallet + originator into the closure
     - Match on `Event::Custom(name)`:
       - `"authenticationSuccess"` → set `connected = true`, send `true` on auth oneshot
       - `"authenticationFailed"` → send `false` on auth oneshot
       - Names starting with `"sendMessage-"` → extract room event key, look up callback in subscriptions map. Parse `Payload::Text` first element as `ServerPeerMessage`. Map to `PeerMessage` (fill `recipient` from room ID identity_key prefix, `message_box` from room ID suffix). **Decrypt body inline** using `try_decrypt_message(&wallet, &body, &sender, originator.as_deref()).await` — the on_any callback is async so this preserves message ordering. Call the subscription callback with the decrypted `PeerMessage`.
       - Names starting with `"sendMessageAck-"` → look up and remove oneshot from `pending_acks`, parse status, send `true` for "success" / `false` otherwise
     - Match on `Event::Close` → set `connected = false`
     - Other events → ignore
  4. Configure `.reconnect(true)`, `.reconnect_on_disconnect(true)`.
  5. Call `.connect().await`.
  6. Emit `"authenticated"` event with `json!({"identityKey": identity_key})`.
  7. Wait for auth signal with `tokio::time::timeout(Duration::from_secs(5), auth_rx)`.
  8. On timeout or `false` → return `Err(MessageBoxError::WebSocket("authentication failed or timed out"))`.
  9. Return `Ok(Self { client, subscriptions, pending_acks, joined_rooms, connected })`.

  CRITICAL: The `on_any` callback captures `wallet.clone()` and `originator.clone()` at connect time. Decryption happens **inside the async callback** (not via `tokio::spawn`) — this preserves message ordering since `rust_socketio` serializes `on_any` calls through its internal mutex.

  CRITICAL: When parsing the room ID to extract `recipient` and `message_box` for `PeerMessage`, split on the first `-`. The room event is `sendMessage-{identityKey}-{messageBox}`, so split after `sendMessage-` then split the remainder at the last `-` to separate identity key from message box name. Edge case: message box names may contain hyphens, but identity keys are hex-only (no hyphens), so split at the boundary between hex chars and the first non-hex-following-a-hyphen, OR simpler: store the `(identity_key, message_box)` pair alongside the callback in the subscriptions map.

- **`pub fn is_connected(&self) -> bool`**: Read `AtomicBool` with `Ordering::SeqCst`.

- **`pub async fn join_room(&self, room_id: &str) -> Result<(), MessageBoxError>`**: Check `joined_rooms` HashSet — skip if already joined. Emit `"joinRoom"` event with `json!(room_id)`. Add to `joined_rooms`.

- **`pub async fn leave_room(&self, room_id: &str) -> Result<(), MessageBoxError>`**: Remove from `joined_rooms` and `subscriptions`. Emit `"leaveRoom"` event with `json!(room_id)`.

- **`pub async fn subscribe(&self, event_key: String, callback: Arc<dyn Fn(PeerMessage) + Send + Sync>)`**: Insert into subscriptions map under `event_key` (e.g., `"sendMessage-{room_id}"`).

- **`pub async fn emit_send_message(&self, payload: serde_json::Value, ack_key: String, ack_tx: oneshot::Sender<bool>) -> Result<(), MessageBoxError>`**: Register ack_tx in pending_acks under ack_key. Emit `"sendMessage"` event with payload. Map rust_socketio errors to `MessageBoxError::WebSocket`.

- **`pub async fn remove_pending_ack(&self, key: &str)`**: Remove entry from pending_acks (for timeout cleanup — Pitfall 7).

- **`pub async fn disconnect(&self) -> Result<(), MessageBoxError>`**: Set `connected = false`. Call `self.client.disconnect().await`. Clear `subscriptions`, `joined_rooms`, drain `pending_acks` (send `false` to all remaining oneshots).

NOTES on reconnect: `rust_socketio` handles reconnect with exponential backoff internally when `.reconnect(true)` is set. This satisfies WS-05's reconnect requirement without manual backoff code. After reconnect, the `on_any` callback is preserved (it's registered at build time), but `joined_rooms` are NOT automatically re-subscribed. The `ensure_ws_connected` method in client.rs handles reconnection at a higher level — if `is_connected()` returns false, it creates a new connection. Room re-subscription is the caller's responsibility (matches TS behavior).

**2. Add `ensure_ws_connected` private method:**

```rust
async fn ensure_ws_connected(&self) -> Result<(), MessageBoxError> {
    let mut guard = self.ws_state.lock().await;
    if guard.as_ref().map(|ws| ws.is_connected()).unwrap_or(false) {
        return Ok(());
    }
    // Get identity key for auth handshake
    let identity_key = self.get_identity_key().await?;
    // Build WebSocket URL: socket.io connects to the same host
    // (rust_socketio handles /socket.io/ path internally)
    let ws_url = self.host().to_string();
    let ws = crate::websocket::MessageBoxWebSocket::connect(
        &ws_url,
        &identity_key,
        self.wallet.clone(),
        self.originator.clone(),
    ).await?;
    *guard = Some(ws);
    Ok(())
}
```

NOTE: Unlike the original plan, we do NOT replace http→ws in the URL. `rust_socketio` handles protocol negotiation internally — it starts with HTTP polling then upgrades to WebSocket. We pass the same base URL (e.g., `https://messagebox.babbage.systems`).

**3. Add `listen_for_live_messages` public method:**

```rust
pub async fn listen_for_live_messages(
    &self,
    message_box: &str,
    on_message: Arc<dyn Fn(PeerMessage) + Send + Sync>,
) -> Result<(), MessageBoxError> {
    let identity_key = self.get_identity_key().await?;
    let room_id = format!("{identity_key}-{message_box}");
    let event_key = format!("sendMessage-{room_id}");

    self.ensure_ws_connected().await?;

    // Acquire lock, join room and subscribe, then drop lock
    let guard = self.ws_state.lock().await;
    if let Some(ref ws) = *guard {
        ws.join_room(&room_id).await?;
        ws.subscribe(event_key, on_message).await;
    }
    Ok(())
}
```

NOTE: Auto-decryption happens in the `on_any` callback inside `MessageBoxWebSocket` (set up at connect time) — NOT here. The callback the user passes receives already-decrypted `PeerMessage` structs. This is different from the original plan which wrapped the callback with a `tokio::spawn` decryption step. The new approach preserves message ordering.

**4. Add `send_live_message` public method:**

```rust
pub async fn send_live_message(
    &self,
    recipient: &str,
    message_box: &str,
    body: &str,
) -> Result<String, MessageBoxError> {
    // Check if WS is available — don't encrypt yet if we'll fall back
    let ws_available = {
        let guard = self.ws_state.lock().await;
        guard.as_ref().map(|ws| ws.is_connected()).unwrap_or(false)
    };

    if !ws_available {
        // HTTP fallback — send_message handles its own encryption
        return self.send_message(recipient, message_box, body).await;
    }

    // Encrypt + generate message ID (same as send_message)
    let encrypted = crate::encryption::encrypt_body(
        self.wallet(), body, recipient, self.originator()
    ).await?;
    let message_id = crate::encryption::generate_message_id(
        self.wallet(), body, recipient, self.originator()
    ).await?;

    let room_id = format!("{recipient}-{message_box}");
    let ack_key = format!("sendMessageAck-{room_id}");

    let (ack_tx, ack_rx) = tokio::sync::oneshot::channel::<bool>();

    // Build payload as JSON value for socket.emit
    let payload = serde_json::json!({
        "roomId": room_id,
        "message": {
            "messageId": message_id,
            "recipient": recipient,
            "body": encrypted
        }
    });

    // Emit — acquire lock briefly
    {
        let guard = self.ws_state.lock().await;
        if let Some(ref ws) = *guard {
            ws.emit_send_message(payload, ack_key.clone(), ack_tx).await?;
        }
    }

    // Await ack with 10-second timeout
    match tokio::time::timeout(std::time::Duration::from_secs(10), ack_rx).await {
        Ok(Ok(true)) => Ok(message_id),
        _ => {
            // Pitfall 7: clean up pending ack on timeout
            let guard = self.ws_state.lock().await;
            if let Some(ref ws) = *guard {
                ws.remove_pending_ack(&ack_key).await;
            }
            drop(guard);
            // Fall back to HTTP — send_message handles its own encryption
            self.send_message(recipient, message_box, body).await
        }
    }
}
```

NOTE: Encryption only happens if WS is available. On HTTP fallback, `send_message` encrypts independently. This avoids wasted work in the common fallback path.
